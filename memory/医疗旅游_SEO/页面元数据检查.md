### 文章页面标准模板

每篇文章应包含以下完整元数据：

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    
    <!-- 基础元数据 -->
    <title>[文章标题] | China Hospitals Guide</title>
    <meta name="description" content="[150-160字符描述]">
    <meta name="keywords" content="[3-5个关键词]">
    <meta name="author" content="China Hospitals Guide">
    <meta name="robots" content="index, follow">
    <link rel="canonical" href="https://chinahospitalsguide.com/blog/[slug].html">
    
    <!-- Open Graph -->
    <meta property="og:title" content="[文章标题]">
    <meta property="og:description" content="[描述]">
    <meta property="og:type" content="article">
    <meta property="og:url" content="https://chinahospitalsguide.com/blog/[slug].html">
    <meta property="og:image" content="https://chinahospitalsguide.com/images/[article-image].jpg">
    <meta property="article:published_time" content="2026-03-11T00:00:00+08:00">
    <meta property="article:modified_time" content="2026-03-11T00:00:00+08:00">
    <meta property="article:author" content="China Hospitals Guide">
    <meta property="article:section" content="Medical Tourism">
    <meta property="article:tag" content="[标签1]">
    <meta property="article:tag" content="[标签2]">
    
    <!-- Twitter Card -->
    <meta name="twitter:card" content="summary_large_image">
    <meta name="twitter:title" content="[文章标题]">
    <meta name="twitter:description" content="[描述]">
    <meta name="twitter:image" content="https://chinahospitalsguide.com/images/[article-image].jpg">
    
    <!-- 结构化数据 -->
    <script type="application/ld+json">
    {
        "@context": "https://schema.org",
        "@type": "BlogPosting",
        "headline": "[文章标题]",
        "description": "[描述]",
        "image": "https://chinahospitalsguide.com/images/[article-image].jpg",
        "author": {
            "@type": "Organization",
            "name": "China Hospitals Guide"
        },
        "publisher": {
            "@type": "Organization",
            "name": "China Hospitals Guide",
            "logo": {
                "@type": "ImageObject",
                "url": "https://chinahospitalsguide.com/logo.png"
            }
        },
        "datePublished": "2026-03-11",
        "dateModified": "2026-03-11",
        "mainEntityOfPage": {
            "@type": "WebPage",
            "@id": "https://chinahospitalsguide.com/blog/[slug].html"
        }
    }
    </script>
</head>
```
